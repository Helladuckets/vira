# Vira runtime dependencies.
#
# Install into the EXISTING .venv only:
#     .venv/bin/pip install -r requirements.txt
#
# NEVER rebuild the venv (no `python3 -m venv`): the macOS Full Disk Access
# grant that lets the watcher read chat.db is tied to the current
# .venv/bin/python binary — a rebuilt venv silently loses it (see README).
# The media-index extras (torch, transformers, insightface, mlx-whisper,
# pyobjc) are installed the same way, on demand; they are deliberately not
# pinned here.
fastapi
uvicorn[standard]
# Pairing QR for the Android companion app (server/companion.py). Pure
# python, zero transitive deps; absent at runtime the pairing card just
# falls back to the copyable payload.
segno
# Live two-way agent sessions (steering + per-tool permission gating).
# Optional at runtime: if this import fails, Vira falls back to the legacy
# one-shot subprocess path and still boots.
claude-agent-sdk==0.2.115
# The vault engine behind the Brain window (brings numpy). Pinned by tag:
# a Vira commit that needs a newer engine bumps this pin in the SAME
# commit, so code and engine always travel together (the in-app updater
# installs dependency changes on update). On a dev machine with a local
# qocha checkout, use an editable install (pip install -e ../qocha) — the
# updater detects editable installs and never overwrites them.
qocha @ git+https://github.com/Helladuckets/qocha@v0.3.0
